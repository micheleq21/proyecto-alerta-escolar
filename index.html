// Verificar si ya existe configuración if (!localStorage.getItem('usuarioAlerta')) { window.location.href = 'config.html'; } // Al apretar el botón, ahora leemos los datos guardados document.getElementById('boton-panico').addEventListener('click', async () => { const datosUsuario = JSON.parse(localStorage.getItem('usuarioAlerta')); // Aquí es donde más adelante añadiremos la geolocalización console.log("Enviando alerta para:", datosUsuario); try { const response = await fetch('/api/alerta', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify(datosUsuario) // Enviamos todos los datos al servidor }); const data = await response.json(); alert("Alerta enviada para: " + datosUsuario.nombre); } catch (error) { alert("Hubo un error al enviar la alerta"); } });
